import API from "../api.js?v=9999"; function getUserIdFromQuery() { const params = new URLSearchParams(window.location.search); return params.get("user"); } function buildNav(userId) { const base = "/app/pages/skater"; const links = [ { href: `${base}/dashboard.html?user=${userId}`, label: "Dashboard" }, { href: `${base}/profile.html?user=${userId}`, label: "Profile" }, { href: `${base}/shows.html?user=${userId}`, label: "Shows" }, { href: `${base}/lessons.html?user=${userId}`, label: "Lessons" }, { href: `${base}/businesses.html?user=${userId}`, label: "Businesses" }, { href: `${base}/music.html?user=${userId}`, label: "Music" } ]; const nav = document.getElementById("skater-nav"); nav.innerHTML = ""; const currentPath = window.location.pathname; links.forEach(link => { const a = document.createElement("a"); a.href = link.href; a.textContent = link.label; if (link.href.includes("shows.html") && currentPath.includes("shows.html")) { a.classList.add("rs-dash-nav-active"); } nav.appendChild(a); }); } const userId = getUserIdFromQuery(); async function loadShows() { const listEl = document.getElementById("skater-shows-full"); const statusEl = document.getElementById("skater-status"); if (!userId) { statusEl.textContent = "Missing skater ID in URL."; return; } buildNav(userId); statusEl.textContent = "Loading shows…"; const res = await API.get(`/api/skater/shows?user=${encodeURIComponent(userId)}`); if (!res.success) { statusEl.textContent = res.error?.message || "Failed to load shows."; return; } const data = res.data || {}; const shows = Array.isArray(data.shows) ? data.shows : []; listEl.innerHTML = ""; if (shows.length === 0) { listEl.innerHTML = "
  • No shows yet.
    • "; } else { shows.forEach(show => { const li = document.createElement("li"); li.textContent = `${show.title} — ${show.date} — ${show.venue || ""}`; listEl.appendChild(li); }); } statusEl.textContent = ""; } loadShows();
